namespace = gambling_selection
namespace = gambling_flavour

country_event = {
	id = gambling_selection.1
	title = gambling_selection.1.title
	desc = gambling_selection.1.desc
	picture = GOV_RUINS_CLEAR_eventPicture
	
	is_triggered_only = yes

	option = {
		name = gambling_selection.1.option.a
        
        if = {
            limit = {
                has_gambling_allowed = yes
            }
            custom_tooltip = ALREADY_ALLOW_GAMBLING
        }
        else = {
            set_gambling_allowed = yes
			add_dip_power = -100
            custom_tooltip = GAMBLING_ALLOWED_EXPLANATION
            add_civil_unrest_3 = yes
        }
	}
	option = {
		name = gambling_selection.1.option.b
        
        if = {
            limit = {
                has_gambling_licensed = yes
            }
            custom_tooltip = ALREADY_LICENSE_GAMBLING
        }
        else = {
            set_gambling_licensed = yes
			add_dip_power = -100
            custom_tooltip = GAMBLING_LICENSED_EXPLANATION
            add_civil_unrest_3 = yes
        }
	}
	option = {
		name = gambling_selection.1.option.c
		
        if = {
            limit = {
                has_gambling_forbidden = yes
            }
            custom_tooltip = ALREADY_FORBID_GAMBLING
        }
        else = {
            set_gambling_forbidden = yes
			add_dip_power = -100
            custom_tooltip = GAMBLING_FORBIDDEN_EXPLANATION
            add_civil_unrest_3 = yes
        }
	}
	option = {
		name = event_exit
	}
}

	id = gambling_flavour.1
	title = gambling_flavour.1.title
	desc = gambling_flavour.1.desc
	picture = STRIP_CASINO_eventPicture
	
    trigger = {
        OR = {
            has_gambling_allowed = yes
            has_gambling_licensed = yes
        }
    }
    
	mean_time_to_happen = {
		months = 120
        
        modifier = {
            factor = 0.5
            
            has_gambling_allowed = yes
        }
	}
    
    option = {
		name = gambling_flavour.1.option.c
		
        add_years_of_income = 1
	}
}
