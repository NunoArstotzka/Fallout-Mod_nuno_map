	title = narcotics_selection.1.title
	desc = narcotics_selection.1.desc
	picture = GOV_RUINS_CLEAR_eventPicture
	
	is_triggered_only = yes

	option = {
		name = narcotics_selection.1.option.a

            custom_tooltip = NARCOTICS_ALLOWED_EXPLANATION

        }
	}
	option = {
		name = narcotics_selection.1.option.b
        
        if = {
            limit = {
                has_narcotics_licensed = yes

			add_dip_power = -100
            custom_tooltip = NARCOTICS_LICENSED_EXPLANATION
            add_civil_unrest_3 = yes
        }
	}
	option = {
		name = narcotics_selection.1.option.c
		
        if = {
            limit = {
                has_narcotics_forbidden = yes
            }
            custom_tooltip = ALREADY_FORBID_NARCOTICS
        }
        else = {
            set_narcotics_forbidden = yes
			add_dip_power = -100
            custom_tooltip = NARCOTICS_FORBIDDEN_EXPLANATION
            add_civil_unrest_3 = yes
        }
	}
	option = {
