	title = slavery_selection.1.title
	desc = slavery_selection.1.desc
	picture = GOV_RUINS_CLEAR_eventPicture

	}

		name = slavery_selection.1.option.b

			add_dip_power = -100
            custom_tooltip = SLAVERY_LICENSED_EXPLANATION
            add_civil_unrest_3 = yes
        }
	}
	option = {
		name = slavery_selection.1.option.c
		
        if = {
            limit = {
                has_slavery_forbidden = yes
            }
            custom_tooltip = ALREADY_FORBID_SLAVERY
        }
        else = {
            set_slavery_forbidden = yes
			add_dip_power = -100
            custom_tooltip = SLAVERY_FORBIDDEN_EXPLANATION
            add_civil_unrest_3 = yes
        }
	}
