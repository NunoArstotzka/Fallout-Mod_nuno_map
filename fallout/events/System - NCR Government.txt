# Unify the Republic - Emperor
country_event = {
	id = ncr_government.3
	title = ncr_government.3.title
	desc = ncr_government.3.desc
	picture = NCR_STATUE_eventPicture

	major = yes
	is_triggered_only = yes
	
	option = {
		name = ncr_government.3.option.a
        
		change_tag = NCR
        change_government_effect = {
            TYPE = ncr_government
            REFORM = ncr_reform
            LEGACY = ncr_legacy
        }
        
		every_province = {
			limit = {
				is_part_of_hre = yes
				OR = {
					owned_by = ROOT
					owner = { is_neighbor_of = ROOT }
				}
				NOT = { is_core = ROOT }
			}
			add_core = emperor
		}
		every_country = {
			limit = {
				is_part_of_hre = yes
				is_emperor = no
				num_of_cities = 1
			}
			country_event = { id = ncr_government.4 }
		}
		dismantle_hre = yes
		set_government_rank = 3
		add_adm_power = 250
	}
}
