#-------------------------------------
# Setups
#-------------------------------------
add_base_pops = {
    # Low Dev
    every_owned_province = {
        limit = {
            NOT = { development = 8 }
        }
        add_low_pop_$TYPE$ = yes
    }
    
    # Med Dev
    every_owned_province = {
        limit = {
            development = 8
            NOT = { development = 16 }
        }
        add_med_pop_$TYPE$ = yes
    }
    
    # High Dev
    every_owned_province = {
        limit = {
            development = 16
        }
        add_high_pop_$TYPE$ = yes
    }
}

add_pops_to_terrain = { 
    # Low Dev
    every_owned_province = {
        limit = {
            has_terrain = $TERRAIN$
            NOT = { development = 8 }
        }
        add_low_pop_$TYPE$ = yes
    }
    
    # Med Dev
    every_owned_province = {
        limit = {
            has_terrain = $TERRAIN$
            development = 8
            NOT = { development = 16 }
        }
        add_med_pop_$TYPE$ = yes
    }
    
    # High Dev
    every_owned_province = {
        limit = {
            has_terrain = $TERRAIN$
            development = 16
        }
        add_high_pop_$TYPE$ = yes
    }
}
