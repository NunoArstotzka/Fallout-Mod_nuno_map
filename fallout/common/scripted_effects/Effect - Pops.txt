boost_majority_status = {
    if = {
        limit = {
            has_province_modifier = maligned_majority_$TYPE$
        }
        set_majority_accepted = { TYPE = $TYPE }
    }
    if = {
        limit = {
            has_province_modifier = accepted_majority_$TYPE$
        }
        set_majority_integrated = { TYPE = $TYPE }
    }
}
reduce_majority_status = {
    if = {
        limit = {
            has_province_modifier = accepted_majority_$TYPE$
        }
        set_majority_maligned = { TYPE = $TYPE }
    }
    if = {
        limit = {
            has_province_modifier = integrated_majority_$TYPE$
        }
        set_majority_accepted = { TYPE = $TYPE }
    }
}
set_majority_maligned = {
    hidden_effect = { 
        remove_province_modifier = accepted_majority_$TYPE$
        remove_province_modifier = integrated_majority_$TYPE$
    }
    add_permanent_province_modifier = {
        name = maligned_majority_$TYPE$
        duration = -1
    }
}
set_majority_accepted = {
    hidden_effect = { 
        remove_province_modifier = maligned_majority_$TYPE$
        remove_province_modifier = integrated_majority_$TYPE$
    }
    add_permanent_province_modifier = {
        name = accepted_majority_$TYPE$
        duration = -1
    }
}
set_majority_integrated = {
    hidden_effect = { 
        remove_province_modifier = accepted_majority_$TYPE$
        remove_province_modifier = maligned_majority_$TYPE$
    }
    add_permanent_province_modifier = {
        name = integrated_majority_$TYPE$
        duration = -1
    }
}
