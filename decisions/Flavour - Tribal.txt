		}
		allow =  {

			any_owned_province = {
                has_province_modifier = pf_vault
            }
		}

			change_government_via_reform = vault_reform
            add_stability = -6
		}
		ai_will_do = {
			factor = 1
		}
	}
    
	assemble_a_raiding_party = {
		potential = {

			NOT = { has_country_modifier = assembled_raiding_party }
		}
		allow =  {

			is_at_war = yes
		}
	  	effect = {
			capital_scope = {
				infantry = ROOT
				cavalry = ROOT
			}
			add_country_modifier = { 
				name = assembled_raiding_party
				duration = 3650
				hidden = yes
			}
		}
		ai_will_do = {
			factor = 1
		}
	}
	
	construct_chem_lab = {
		potential = {
			government = tribal
			NOT = { has_country_modifier = chem_lab }

			}
		}
		allow =  {
            government = tribal
			treasury = 500
			OR = {
				has_narcotics_allowed = yes
				has_narcotics_licensed = yes
			}
		}
	  	effect = {
			add_treasury = -500

			factor = 1
