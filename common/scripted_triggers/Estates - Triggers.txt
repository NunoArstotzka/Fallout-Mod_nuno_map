has_generic_estates = {
    NOT = { has_reform = think_tank_reform }
    NOT = { has_reform = zetan_reform }
    NOT = { has_reform = institute_reform }
    NOT = { has_reform = tribe_reform }
    NOT = { has_reform = commune_reform }
    NOT = { has_reform = raider_reform }
    NOT = { has_reform = super_mutant_reform }
    NOT = { has_reform = swarm_reform }
    NOT = { has_reform = wolf_pack_reform }
    NOT = { has_reform = great_khan_reform }
}

has_think_tank_estates = {
    has_reform = think_tank_reform
}

has_institute_estates = {
    has_reform = institute_reform
}

has_tribal_estates = {
    OR = {
        has_reform = tribe_reform
        has_reform = commune_reform
    }
}

has_raider_estates = {
    OR = {
        has_reform = wolf_pack_reform
        has_reform = raider_reform
    }
}

has_super_mutant_estates = {
    has_reform = super_mutant_reform
}

has_khan_estates = {
    has_reform = great_khan_reform
}
