feral_ghoul_rebels = {

	area = nation 					# Stays within the nation
	government = any			
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	unit_transfer = yes
	gfx_type = ghoul_gfx
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = no
	
	artillery = 0.0
	infantry = 0.8
	cavalry = 0.2
	
	morale = 0.9
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = yes
	handle_action_build_core = yes	
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 10
			culture_group = ghoul_group
            owner = {
        		NOT = { culture_group = ghoul_group }
         	}
		}
		modifier = {
			factor = 2
			religion = bright_brotherhood # Large population of sentient ghouls attract ferals
		}
		modifier = {
			factor = 1.2
			religion = renewal # Large population of sentient ghouls attract ferals
		}
        modifier = {
			factor = 0.25
			culture_group = ghoul_group
           	owner = {
				culture_group = ghoul_group
       		}
		}
        
		modifier = {
			factor = 0
			NOT = {
				culture_group = ghoul_group
				owner = { religion = bright_brotherhood }
				owner = { religion = renewal }
				religion = bright_brotherhood
				religion = renewal
			}
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 0.6
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
		modifier = {
			factor = 1.1
			NOT = {
				religion = bright_brotherhood
				religion = renewal
			}
			owner = {
				OR = {
					religion = renewal
					religion = bright_brotherhood
				}
			}
		}
		modifier = {
			factor = 2
			OR = {
				religion = bright_brotherhood
				religion = renewal
			}
			owner = {
				NOT = {
					religion = renewal
					religion = bright_brotherhood
				}
			}
		}
		modifier = {
			factor = 10
			culture_group = ghoul_group
			owner = {
				NOT = { culture_group = ghoul_group }
			}
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
 	}
	
	siege_won_effect = {
        if = { limit = { has_global_flag = enable_common_logging } log = "raiders.txt - siege_won_effect - [Root.GetName]" }
		add_province_modifier = {
			name = damaged_population
			duration = 3650
		}
		add_colonysize = -400
        add_devastation = 5
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "raiders_rebels_demand"
	
 	# Country Scope
	demands_enforced_effect = {
		if = { limit = { has_global_flag = enable_common_logging } log = "raiders.txt - demands_enforced_effect - [Root.GetName]" }
		add_prestige = -50
		if = {
			limit = {
				OR = {
					is_true_monarchy = yes
					is_hereditary_government = yes
					government = administrative_settlement
					government = diplomatic_settlement
					government = military_Settlement
				}
            }
			add_legitimacy = -50
		}
		if = { limit = { is_republic_government = yes } add_republican_tradition = -50 }
		if = { limit = { is_tribe_government = yes } add_horde_unity = -50 }
		if = { limit = { is_theocracy_government = yes } add_devotion = -50 }
		add_stability = -2
	}
}
