add_mutual_opinion_modifier_effect = {
	add_opinion = {
		who = $scope_country$
		modifier = $opinion_modifier$
	}
	reverse_add_opinion = {
		who = $scope_country$
		modifier = $opinion_modifier$
	}
}

add_or_extend_province_modifier_effect = {
	if = {
		limit = {
			has_province_modifier = $MODIFIER$
		}
		extend_province_modifier = { name = $MODIFIER$ duration = $DURATION$ }
	}
	else = {
		add_province_modifier = { name = $MODIFIER$ duration = $DURATION$ }
	}
}

grant_immortality = {
    add_ruler_modifier = {
        name = immortality
        duration = -1
    }
}

nuke_province = {
    cede_province = ---
    remove_core = ROOT
    add_province_modifier = {
        name = nuked_province
        duration = 36500
    }
}
