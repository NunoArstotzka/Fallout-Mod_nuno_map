# Slavery
has_slavery_allowed = { 
    custom_trigger_tooltip = {
        tooltip = SLAVERY_ALLOWED_TT
        has_country_flag = policy_allows_slavery
    }
}
has_slavery_forbidden = { 
    custom_trigger_tooltip = {
        tooltip = SLAVERY_FORBIDDEN_TT
        NOT = { has_country_flag = policy_allows_slavery }
    }
}
